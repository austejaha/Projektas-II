\hypertarget{index_veikimas_sec}{}\doxysection{Veikimo principas}\label{index_veikimas_sec}

\begin{DoxyItemize}
\item Programa skaičiuoja studentų galutinį balą, naudojant vidurkį arba medianą.
\item Vartojas gali pasirinkti ar duomenis norės įvesti pats, ar duomenys bus nuskaityti.
\item Jei vartotojas pasirenka duomenis įvesti pats, jis taip pat turi nurodyti kiek namų darbų pažymių norės įvesti.
\item Namų darbų pažymius ir egzamino balą vartotojas gali įvesti ranka arba pažymėti, jog jie būtų sugeneruoti automatiškai.
\item Suskaičiavus studentų galutinį balą programa, pagal vartotojo pasirinktą strategiją, surūšiuoja studentus į „\+Genijus\char`\"{} ir „\+Vargšus\char`\"{}.
\item Rezultatai išvedami į atskirus tekstinius failus. 
\end{DoxyItemize}
